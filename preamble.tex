\def\gostopemph#1{%
  \sbox{\titlebox}{\vphantom{gl}#1}%
  \hbox to \wd\titlebox{%
    \rlap{\textcolor{gostopred}{\rule[-.37em]{\wd\titlebox}{.65em}}}%
    \usebox{\titlebox}%
  }%
}

\newcommand\mysymbol{%
  \begin{tikzpicture}[xscale=0.85]
    \draw (1.75ex,0) node[star, star points=5, star point ratio=2.25, inner sep=1.3pt,anchor=outer point 3,fill=gostopred] {};          
  \end{tikzpicture}%
}
\newcommand\mysymboli{%
  \begin{tikzpicture}[xscale=0.85]
    \draw (1.75ex,0) node[star, star points=5, star point ratio=2.25, inner sep=1.3pt,anchor=outer point 3,fill=color0] {};          
  \end{tikzpicture}%
}
\newcommand\mysymbolii{%
  \begin{tikzpicture}[xscale=0.85]
    \draw (1.75ex,0) node[star, star points=5, star point ratio=2.25, inner sep=1.3pt,anchor=outer point 3,fill=color2] {};          
  \end{tikzpicture}%
}
